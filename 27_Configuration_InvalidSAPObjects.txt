SELECT NULL OWNER, NULL OBJECT_TYPE, NULL NUM_INVALID_OBJECTS
  FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL OBJECT_TYPE, NULL NUM_INVALID_OBJECTS
  FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    'SAP%' OWNER
  FROM
    DUAL
)
SELECT
  O.OWNER,
  O.OBJECT_TYPE,
  TO_CHAR(COUNT(*), 999999999999999990) NUM_INVALID_OBJECTS
FROM
  BASIS_INFO BI,
  DBA_OBJECTS O
WHERE
  O.OWNER LIKE BI.OWNER AND
  O.STATUS = 'INVALID'
GROUP BY
  O.OWNER,
  O.OBJECT_TYPE
ORDER BY
  O.OWNER,
  O.OBJECT_TYPE
));

SELECT NULL OWNER, NULL OBJECT_NAME, NULL OBJECT_TYPE, NULL LAST_DDL_TIME,
  NULL STATUS FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL OWNER, NULL OBJECT_NAME, NULL OBJECT_TYPE, NULL LAST_DDL_TIME,
  NULL STATUS FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
SELECT
  OWNER,
  OBJECT_NAME,
  OBJECT_TYPE,
  TO_CHAR(LAST_DDL_TIME, 'dd.mm.yyyy hh24:mi:ss') LAST_DDL_TIME,
  STATUS
FROM
  DBA_OBJECTS
WHERE
  OWNER LIKE 'SYS%' AND
  STATUS = 'INVALID'
ORDER BY
  OBJECT_NAME
));
